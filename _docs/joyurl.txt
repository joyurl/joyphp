#管理员账号
CREATE TABLE `adm_user` (
  `user_id` int(11) NOT NULL AUTO_INCREMENT COMMENT '会员编号',
  `user_name` varchar(60) DEFAULT NULL COMMENT '登录账号',
  `user_nickname` varchar(60) DEFAULT NULL COMMENT '账号名称或备注',
  `user_level` tinyint(1) NOT NULL DEFAULT '0' COMMENT '管理员级别{0 暂停的 4 删除的 5 一般管理员 9系统管理员}',
  `user_key` int(6) NOT NULL DEFAULT '0' COMMENT '加密附加字串',
  `user_password` varchar(33) NOT NULL COMMENT '登录密码{使用加密附加字串及原始密码MD5加密}',
  `user_adddate` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '账号添加时间',
  `user_logdate` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '最后一次登录时间',
  `user_logtimes` int(10) NOT NULL DEFAULT '0' COMMENT '账号登录次数',
  `user_errdate` date NOT NULL DEFAULT '0000-00-00' COMMENT '登录日期',
  `user_errtimes` int(10) NOT NULL DEFAULT '0' COMMENT '当日密码错误次数',
  PRIMARY KEY (`user_id`),
  UNIQUE KEY `user_name` (`user_name`)
) ENGINE=MyISAM AUTO_INCREMENT=1001 DEFAULT CHARSET=utf8 COMMENT='管理员账号';

#权限设置
CREATE TABLE `adm_user_limit` (
  `user_id` int(11) NOT NULL COMMENT '会员编号',
  `user_limit` varchar(600) NOT NULL DEFAULT '' COMMENT '权限设置{格式为:[name]权限符。权限符说明:*全部权限;@全部只读;_只读自已的;&可增加和读自己的;#可增加和改自己的}',
  PRIMARY KEY (`user_id`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='管理员权限设置';

#登录日志
CREATE TABLE `adm_user_login` (
  `login_id` int(10) NOT NULL AUTO_INCREMENT COMMENT '自增编号',
  `user_id` int(10) NOT NULL DEFAULT '0' COMMENT '会员编号',
  `user_date` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '登录时间',
  `user_ip` varchar(15) DEFAULT NULL COMMENT '客户端IP',
  `user_port` int(5) NOT NULL DEFAULT '0' COMMENT '客户端端口',
  `server_ip` varchar(15) NOT NULL DEFAULT '0' COMMENT '服务端IP',
  `server_port` int(5) NOT NULL DEFAULT '0' COMMENT '服务端端口',
  PRIMARY KEY (`login_id`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='管理员登录日志';

#管理日志（记录修改前的内容）
CREATE TABLE `adm_change_log` (
  `log_id` int(10) NOT NULL AUTO_INCREMENT COMMENT '自增编号',
  `user_id` int(10) NOT NULL DEFAULT '0' COMMENT '会员编号',
  `user_date` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '操作时间',
  `user_ip` varchar(15) DEFAULT NULL COMMENT '客户端IP',
  `user_port` int(5) NOT NULL DEFAULT '0' COMMENT '客户端端口',
  `change_name` varchar(15) NOT NULL DEFAULT '0' COMMENT '操作模块名称',
  `change_action` varchar(15) NOT NULL DEFAULT '0' COMMENT '操作名称',
  `change_cont` text DEFAULT NULL COMMENT '修改前的内容',
  PRIMARY KEY (`log_id`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='管理员操作日志';

#主配置表
CREATE TABLE `adm_config` (
  `conf_name` varchar(50) NOT NULL DEFAULT '' COMMENT '英文名称{模板中显示的标签名}',
  `conf_title` varchar(60) NOT NULL DEFAULT '' COMMENT '中文显示名称',
  `conf_value` varchar(250) NOT NULL DEFAULT '' COMMENT '设置值',
  `conf_type` tinyint(1) DEFAULT '0' COMMENT '类型{值及含义:0系统(不允许删除) 1 自定义(允许删除的)}',
  `conf_group` varchar(5) DEFAULT NULL COMMENT '设置分组',
  `conf_sort` int(5) NOT NULL DEFAULT '0' COMMENT '设置排序',
  PRIMARY KEY (`conf_name`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='主配置表';

#模型配置 presenter

  `model_sort` tinyint(2) DEFAULT '0', #排序
  PRIMARY KEY (`data_name`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='模型设置';

  PRIMARY KEY (`data_name`)
) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='数据表设置';
